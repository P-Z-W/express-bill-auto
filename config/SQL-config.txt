SELECT
    p.waybill_id AS 运单号,
    p.created_time,
    p.team_id,
    t.team_name AS 所属团队,
    p.delivery_id,
    CASE
        WHEN p.delivery_id = 7 THEN '圆通'
        WHEN p.delivery_id = 8 THEN '申通'
        WHEN p.delivery_id = 12 THEN '顺丰'
        WHEN p.delivery_id = 15 THEN '中通'
        WHEN p.delivery_id = 30 THEN '京东'
        WHEN p.delivery_id = 102 THEN '极兔'
        WHEN p.delivery_id = 1109 THEN '邮政电商标快'
        ELSE '其他快递'
    END AS 快递类型
FROM
    pre_matched_order p
INNER JOIN team_source t 
    ON p.team_id = t.team_id
WHERE
    p.created_time >= '{START_DATE}'
    AND p.created_time <= '{END_DATE}'
    AND p.waybill_id IS NOT NULL
    AND p.waybill_id <> '';
